┌──────────────────────────────────────────────────────────────────────────┐
│ 💡 KEY TAKEAWAYS                                                          │
├──────────────────────────────────────────────────────────────────────────┤
│  1. ViT5 = Encoder-Decoder model (T5-style) cho tiếng Việt               │
│  2. Base: 220M params, Large: 770M params                                │
│  3. Task prefix BẮT BUỘC: "tóm tắt:", "dịch:", etc.                      │
│  4. Best models: vit5-large-vietnews (summarization)                     │
│                   envit5-base (translation)                              │
│  5. So với Qwen: ViT5 = seq2seq tasks, Qwen = chat/creative             │
│  6. Memory: Base 0.44GB (FP16), Large 1.54GB (FP16)                      │
│  7. Fine-tune dễ dàng với HuggingFace Trainer                             │
│  8. Chạy được trên CPU (chậm), GPUrecommended                           │
└──────────────────────────────────────────────────────────────────────────┘

│  from transformers import AutoTokenizer, AutoModelForSeq2SeqLM           │
│  import torch                                                             │
│                                                                           │
│  # Load                                                                   │
│  tokenizer = AutoTokenizer.from_pretrained("VietAI/vit5-base")          │
│  model = AutoModelForSeq2SeqLM.from_pretrained("VietAI/vit5-base")      │
│  model.to("cuda")  # hoặc "cpu"                                          │
│                                                                           │
│  # Task prefix                                                           │
│  text = "tóm tắt: " + long_text + " </s>"                                │
│                                                                           │
│  # Generate                                                               │
│  inputs = tokenizer(text, return_tensors="pt").to(device)               │
│  outputs = model.generate(                                               │
│      **inputs,                                                           │
│      max_length=100,                                                     │
│      num_beams=4,                                                        │
│      length_penalty=2.0                                                  │
│  )                                                                       │
│                                                                           │
│  # Decode                                                                │
│  result = tokenizer.decode(outputs[0], skip_special_tokens=True)         │
│  print(result)                                                           │
│                                                                           │
